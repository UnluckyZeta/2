import { createClient } from "npm:@supabase/supabase-js@2"; const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type", "Access-Control-Allow-Methods": "POST, OPTIONS", }; const supabase = createClient( Deno.env.get("SUPABASE_URL")!, Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!, ); Deno.serve(async req => { if (req.method === "OPTIONS") { return new Response("ok", { headers: corsHeaders, }); } try { const { from, to, } = await req.json(); const { data: profiles, error } = await supabase .from("profiles") .select("*") .order("created_at", { ascending: false, }); if (error) { throw error; } const engineers = await Promise.all( profiles.map(async engineer => { const { data: logs } = await supabase .from("photo_logs") .select(` id, taken_at, accuracy, speed, heading, altitude `) .eq( "engineer_id", engineer.id, ); const photoCount = logs?.length ?? 0; const lastPhoto = logs?.sort( (a, b) => new Date( b.taken_at, ).getTime() - new Date( a.taken_at, ).getTime(), )[0]; const lastPhotoAt = lastPhoto?.taken_at ?? null; let daysInactive = null; if (lastPhotoAt) { daysInactive = Math.floor( (Date.now() - new Date( lastPhotoAt, ).getTime()) / 86400000, ); } let gpsRisk = "normal"; const suspiciousLog = logs?.some( log => (log.accuracy ?? 999) < 1 && (log.speed ?? 0) === 0 && (log.heading ?? 0) === 0 && (log.altitude ?? 0) === 0, ); if ( suspiciousLog ) { gpsRisk = "high"; } return { ...engineer, photo_count: photoCount, last_photo_at: lastPhotoAt, days_inactive: daysInactive, status: daysInactive === null ? "never" : daysInactive >= 14 ? "inactive" : daysInactive >= 7 ? "warning" : "active", gps_risk: gpsRisk, }; }), ); let photosQuery = supabase .from("photo_logs") .select("*", { count: "exact", head: true, }); if (from) { photosQuery = photosQuery.gte( "taken_at", from, ); } if (to) { photosQuery = photosQuery.lte( "taken_at", `${to} 23:59:59`, ); } const { count: totalPhotos, error: photosError, } = await photosQuery; if (photosError) { throw photosError; } return Response.json( { engineers, totalPhotos: totalPhotos ?? 0, }, { headers: { ...corsHeaders, "Content-Type": "application/json", }, }, ); } catch (e) { return new Response( JSON.stringify({ success: false, error: String(e), }), { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json", }, }, ); } });
